### the formula {#the-formula}

\begin{align\*}
t &= \frac{\bar{d} - \mu\_{o}}{s/\sqrt{n}}\\\\
s &= \sqrt{\frac{1}{n-1}\sum\_{i=1}^{n} (d\_{i} - \bar{d})^{2}}\\\\
\bar{d} &= \text{ mean of differences}
\end{align\*}
